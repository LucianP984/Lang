// Example 4: Control flow (if/while statements)
// If statement
x = 10
if (x > 5) {
    print "x is greater than 5"
} else {
    print "x is not greater than 5"
}

// If-else chain
y = 3
if (y > 5) {
    print "y is greater than 5"
} else if (y > 0) {
    print "y is positive but not greater than 5"
} else {
    print "y is negative or zero"
}

// While loop
count = 1
while (count <= 5) {
    print count
    count = count + 1
}
print "Loop finished!"

// Nested control flow
i = 1
while (i <= 3) {
    j = 1
    while (j <= 3) {
        print i * j
        j = j + 1
    }
    i = i + 1
}
